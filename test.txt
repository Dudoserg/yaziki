int fuct(int a)
{
  if (a > 1)
  {
    return a * fuct(a - 1);
  }
  else
    return 1;
}

int globali = 0;
int globaln = 0;
double globalpoverty = 0;
double globalx = 0;

double cycle(){
    if( globali < 2 * globaln - 1){
        globali = globali + 1;
        globalpoverty = globalpoverty * globalx;
        cycle();
    }
}

double pover(double x, int n)
{
  double poverty = 1;
  globalpoverty = 1;
  globaln = n;
  globali = 0;
  globalx = x;

  cycle();
  poverty = globalpoverty;

  return poverty;
}

int sgn(int n)
{
  int k = 0;
  k = n % 2;
  if (k == 0)
  {
    return 0 - 1;
  }
  else
    return 1;
}

double expended(double x)
{
  double summ = 0;


    summ = summ + (sgn(1)*pover(x, 1)) / fuct(2 * 1 - 1);
    summ = summ + (sgn(2)*pover(x, 2)) / fuct(2 * 2 - 1);
    summ = summ + (sgn(3)*pover(x, 3)) / fuct(2 * 3 - 1);
    summ = summ + (sgn(4)*pover(x, 4)) / fuct(2 * 4 - 1);
    summ = summ + (sgn(5)*pover(x, 5)) / fuct(2 * 5 - 1);


  return summ;
}
